CREATE VIEW view_name AS

SELECT t.tennant_no , t.t_forenames , t.t_surnames

FROM tennant t

WHERE EXTRACT (HOUR FROM CURRENT_TIME) >= 12

AND EXTRACT (HOUR FROM CURRENT_TIME) < 17;
